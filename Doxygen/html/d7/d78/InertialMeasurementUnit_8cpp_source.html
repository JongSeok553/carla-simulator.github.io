62 void AInertialMeasurementUnit::Tick(float DeltaTime)
63 {
64  Super::Tick(DeltaTime);
65 
66  namespace cg = carla::geom;
67 
68  // Accelerometer measures linear acceleration in m/s2
69  constexpr float TO_METERS = 1e-2;
70 
71  ACarlaWheeledVehicle *vehicle = Cast<ACarlaWheeledVehicle>(GetOwner());
72  const FVector CurrentVelocity = vehicle->GetVehicleForwardSpeed() * vehicle->GetVehicleOrientation();
73  float CurrentSimulationTime = GetWorld()->GetTimeSeconds();
74  FVector FVectorAccelerometer =
75  TO_METERS * (CurrentVelocity - PrevVelocity) / DeltaTime;
76 
77  PrevVelocity = CurrentVelocity;
78 
79  FQuat ImuRotation = GetRootComponent()->GetComponentTransform().GetRotation();
80  FVectorAccelerometer = ImuRotation.UnrotateVector(FVectorAccelerometer);
81 
82  // Cast from FVector to our Vector3D to correctly send the data in m/s2
83  const cg::Vector3D Accelerometer (
84  FVectorAccelerometer.X,
85  FVectorAccelerometer.Y,
86  FVectorAccelerometer.Z
87  );
88 
89  // Gyroscope measures angular velocity in rad/sec
90  const FVector AngularVelocity =
91  carla_GetActorAngularVelocityInRadians(*GetOwner());
92 
93  const FQuat SensorLocalRotation =
94  RootComponent->GetRelativeTransform().GetRotation();
95 
96  const FVector FVectorGyroscope =
97  SensorLocalRotation.RotateVector(AngularVelocity);
98 
99  // Cast from FVector to our Vector3D to correctly send the data in rad/s
100  const cg::Vector3D Gyroscope (
101  FVectorGyroscope.X,
102  FVectorGyroscope.Y,
103  FVectorGyroscope.Z
104  );
105 
106  // Magnetometer: orientation with respect to the North in rad
107  const FVector ForwVect = GetActorForwardVector().GetSafeNormal2D();
108  float Compass = std::acos(FVector::DotProduct(CarlaNorthVector, ForwVect));
109 
110  // Keep the angle between [0, 2pi)
111  if (FVector::CrossProduct(CarlaNorthVector, ForwVect).Z > 0.0f)
112  {
113  Compass = cg::Math::Pi2<float>() - Compass;
114  }
115 
116  auto Stream = GetDataStream(*this);
117  Stream.Send(
118  *this,
119  Accelerometer,
120  Gyroscope,
121  Compass);
122 }
