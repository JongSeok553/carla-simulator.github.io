20  carla::rpc::VehicleControl RssRestrictor::restrictVehicleControl(
21  const carla::rpc::VehicleControl &vehicleControl, const ad_rss::world::AccelerationRestriction &restriction,
22  const ad_rss::world::Velocity &egoVelocity, const carla::rpc::VehiclePhysicsControl &vehiclePhysics) {
23  carla::rpc::VehicleControl restrictedVehicleControl(vehicleControl);
24 
25  // Pretty basic implementation of a RSS restrictor modifying the
26  // VehicleControl according to the given
27  // restrictions. Basic braking and countersteering actions are applied.
28 
29  ad_rss::physics::Acceleration zeroAccel(0.0);
30 
31  float mass = vehiclePhysics.mass;
32  float sumMaxBrakeTorque = 0.f;
33  float radius = 1.f;
34  for (auto const &wheel : vehiclePhysics.wheels) {
35  sumMaxBrakeTorque += wheel.max_brake_torque;
36  radius = wheel.radius;
37  }
38 
39  // do not apply any restrictions when in reverse gear
40  if (!vehicleControl.reverse) {
41 
42  // decelerate
43  if (restriction.longitudinalRange.maximum < zeroAccel) {
44  restrictedVehicleControl.throttle = 0.0f;
45 
46  double sumBrakeTorque = mass * std::fabs(static_cast<double>(restriction.longitudinalRange.minimum)) *
47  radius / 100.0;
48  restrictedVehicleControl.brake =
49  std::min(static_cast<float>(sumBrakeTorque / sumMaxBrakeTorque), 1.0f);
50  }
51 
52  if (restriction.lateralLeftRange.maximum <= ad_rss::physics::Acceleration(0.0)) {
53  if (egoVelocity.speedLat < ad_rss::physics::Speed(0.0)) {
54  // driving to the left
55  if (egoVelocity.speedLon != ad_rss::physics::Speed(0.0)) {
56  double angle = std::atan(egoVelocity.speedLat / egoVelocity.speedLon);
57  restrictedVehicleControl.steer = -1.f * static_cast<float>(angle);
58  }
59  }
60  }
61  if (restriction.lateralRightRange.maximum <= ad_rss::physics::Acceleration(0.0)) {
62  if (egoVelocity.speedLat > ad_rss::physics::Speed(0.0)) {
63  // driving to the right
64  if (egoVelocity.speedLon != ad_rss::physics::Speed(0.0)) {
65  double angle = std::atan(egoVelocity.speedLat / egoVelocity.speedLon);
66  restrictedVehicleControl.steer = -1.f * static_cast<float>(angle);
67  }
68  }
69  }
70  }
71  return restrictedVehicleControl;
72  }
